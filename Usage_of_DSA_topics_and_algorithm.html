struct Node {
    int id;
    int distance;
    bool operator>(const Node& other) const {
        return distance > other.distance;
    }
};

void dijkstra(int start, const vector>& graph) {
    int n = graph.size();
    vector dist(n, INT_MAX);
    priority_queue, greater> pq;

    dist[start] = 0;
    pq.push({start, 0});

    while (!pq.empty()) {
        Node curr = pq.top();
        pq.pop();

        for (int i = 0; i < n; ++i) {
            if (graph[curr.id][i] != 0 && dist[i] > dist[curr.id] + graph[curr.id][i]) {
                dist[i] = dist[curr.id] + graph[curr.id][i];
                pq.push({i, dist[i]});
            }
        }
    }

    cout << "Shortest distances from node " << start << ":\n";
    for (int i = 0; i < n; ++i) {
        cout << "To " << i << ": " << dist[i] << endl;
    }
}
